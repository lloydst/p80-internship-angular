 * components

 * services

import { ServiceWorkerModule } from '@angular/service-worker';
import { environment } from '../environments/environment.prod';
@NgModule({
  declarations: [
    AppComponent,
  ],
  imports: [
    BrowserModule,
    HttpClientModule,
    FormsModule,

    ChannelsModule,
    ComponentsModule,
    AppRoutingModule,
    FileUploadModule,

  providers: [
    DataService,
    NewsService,
    WeatherService,
    FileService
  ],
  bootstrap: [AppComponent]
})
