import { Component, OnInit } from '@angular/core';
import { DataService } from '../../services/data.service';

/**
 * meeting component
 */
@Component({
  selector: 'app-meeting',
  templateUrl: './meeting.component.html'
})
export class MeetingComponent implements OnInit {
  /**
   * for binding
   */
  
  meeting: any = []
  /**
   * for binding
   */
  loginStatus
  /**
   * constructor
   */
  constructor(
    private data: DataService,
    ) { }
  /**
   * on load
   */

  ngOnInit() {
    this.data.getLoggedIn().subscribe(res=>{
      this.loginStatus= res
      this.data.getCalendar().subscribe(res=>{
        this.meeting= [res]
        
      })
    })
    
    this.ifNotLoggedIn()
  }
  /**
   * if the signInUrl is there it means no one is logged in, so a window gets opened to login
   */
  ifNotLoggedIn() {
    setTimeout(() => {
      if(this.loginStatus.signInUrl) {
        var login = window.open(this.loginStatus.signInUrl,'login') 
        // due to cross origin i cant do more then this need to find a way to auto login
      } 
    }, 200);
  }
  
     }

                <ul class="meeting__top-details-attendees-first">
                    <li *ngFor="let email of meeting[0].events[meeting[0].events?.length -1]?.attendees; index as i">

                        <!-- replacing name with address gives a email address instead of a name-->
                    </li>

                </ul>
                <ul class="meeting__top-details-attendees-second">
                    <li *ngFor="let email of meeting[0].events[meeting[0].events.length -1].attendees; index as i">
                        <span *ngIf="i>3 && i <7">{{email.emailAddress.name}}</span><!-- displays name 4-8-->
                    </li>
                </ul>
            </div>
            <p class="meeting__top-details-description"> {{meeting[0].events[meeting[0].events.length -1].subject}} </p>
        </div>

                    <!-- inverting the list returned by the api (returns last meeting first) it might still pose a problem if there are too many meetings-->
